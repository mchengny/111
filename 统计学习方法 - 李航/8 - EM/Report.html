from  matplotlib import pyplot as plt
from scipy import stats
import numpy as np
%matplotlib inline

class GaussianMixture():
    
    
    def __init__(self, k_components=1, max_iter=100, means_init=None, tol=0.01):
        """
        Initialize the parameters of Gaussian Mixture Model
        
        Arguments:
        k_components -- the number of gaussian distributions in model
        max_iter -- the max number of iteration of EM algorithm
        means_init -- the initial means of given model
        means -- the means of each gaussian distribution
        covs -- the covariance matrix of each gaussian distribution
        alpha -- the weights of each gaussian distribution
        log_likelihood -- a list consisted of log likelihood of each step
        tol -- the convergence threshold. EM iterations will stop when the lower bound average gain 
        is below this threshold.
        
        """
        self.k_components = k_components
        self.max_iter = max_iter
        self.means_init = means_init
        self.tol = tol
        
        self.means = None
        self.covs = None
        self.alpha = np.ones(k_components) / k_components
        self.log_likelihood = []
        
        return None
    
    
    def fit(self, X_train, draw=False):
        """
        Fit the model with given data
        
        Arguments:
        X_train -- training dataset, with the shape of [n_samples, n_features]
        draw -- whether to plot the curve of log likelihood during training
        
        """
        # Get the shape of training dataset
        num_x = X_train.shape[0] 
        dim_x = X_train.shape[1]
        # Initialize the means and co-variances
        self.means = self.init_means(X_train)
        self.covs = self.init_covs(X_train)
        # Compute the 1st log likelihood 
        response = self.response_matrix(X_train)
        last_llf = self.compute_log_likelihood(X_train, response)
        
        # Iterations: optimize step by step
        for num in range(self.max_iter):
            # Compute the responsiveness of each point under given distribution
            response = self.response_matrix(X_train)
            # Loop: iterate each distribution
            for k in range(self.k_components):
                
                # Update the means 
                mu_k = np.array([])
                for d in range(dim_x):
                    numerator = np.dot(response[:,k], X_train[:,d])
                    denominator = np.sum(response[:,k])
                    mu = numerator / denominator
                    mu_k = np.append(mu_k, mu)
                self.means[k] = mu_k 
                
                # Update the cov matrix
                cov_matrix = np.empty([dim_x, dim_x])
                for i in range(dim_x):
                    for j in range(dim_x):
                        cov = (X_train[:,i] - mu_k[i]) * (X_train[:,j] - mu_k[j])
                        cov_matrix[i,j] = np.dot(response[:,k], cov) / np.sum(response[:,k])
                self.covs[k] = cov_matrix
            
                # Update the alpha param 
                weight = np.sum(response[:,k]) / num_x 
                self.alpha[k] = weight
            
            # Get the log likelihood of this step
            llf = self.compute_log_likelihood(X_train, response)
            self.log_likelihood.append(llf)
            # End iterating if model has been convergent
            if abs(llf - last_llf) < self.tol:
                break
            else:
                last_llf = llf
                
        # Plot the curve of training log-likelihood
        if draw == True:
            plt.plot(self.log_likelihood)
            
        return None
    
    
    def predict(self, query):
        """
        Predict the result of clustering
        
        Argument:
        query -- the data point waiting for processing
    
        Return:
        label -- the corresponding label of given query
        
        """
        responses = []
        for k in range(self.k_components):
            r = self.response(query, k)
            responses.append(r)
        
        max_prob = max(responses) 
        label = responses.index(max_prob)
        
        return label
            
        
    def response(self, x, k):
        """
        Compute the responsiveness of each point under k-distribution
        
        Arguments:
        x -- training dataset, with the shape of [n_samples, n_features]
        k -- the No. of ordered gaussian distribution
    
        """
        # Define params in need
        numerator = 0
        denominator = 0    
        # Generate the gaussian distribution under given means and covs
        rv = stats.multivariate_normal(self.means[k], self.covs[k])
        numerator = rv.pdf(x) * self.alpha[k]
        # Compute the denominator of formula
        for i in range(self.k_components):
            rv = stats.multivariate_normal(self.means[i], self.covs[i])
            denominator += rv.pdf(x) * self.alpha[i]
            
        return numerator / denominator
        
        
    def response_matrix(self, X_train):
        """
        Compute the responsiveness matrix of each point under different gaussian models
        
        Argument:
        x -- training dataset, with the shape of [n_samples, n_features]

        """
        # Initialize the matrix
        n_samples = len(X_train) 
        k_models = self.k_components
        matrix = np.empty([n_samples, k_models])
        # Compute each value in matrix
        for i in range(n_samples):
            for j in range(k_models):
                matrix[i,j] = self.response(X_train[i], j)
        
        return matrix
    
    
    def init_means(self, x):
        """
        Initilize the means of each gaussian model
        
        Argument:
        x -- training dataset, with the shape of [n_samples, n_features]
        
        """
        # Whether set the init weights as means or randomly generate them
        if self.means_init == None:
            dim_x = x.shape[1]     
            center  = np.mean(x, axis=0)
            return np.random.normal(0,1,[self.k_components, dim_x]) + center
        else:
            return self.means_init
    
    
    def init_covs(self, x):
        """
        Initilize the co-variance of each gaussian model
        
        Argument:
        x -- training dataset, with the shape of [n_samples, n_features]
        
        Return:
        covs -- the cov matrix of training data under current means
        
        """
        # Geth the shape of training dataset
        num_x = x.shape[0]
        dim_x = x.shape[1]
        
        # Compute each co-variance and keep them into covs
        covs = []
        for k in range(self.k_components):
            mu = self.means[k]
            cov_matrix = np.empty([dim_x, dim_x])
            for i in range(dim_x):
                for j in range(dim_x):
                    cov = np.sum((x[:,i] - mu[i]) * (x[:,j] - mu[j])) / num_x
                    cov_matrix[i,j] = cov      
            covs.append(cov_matrix)  
            
        return covs     
    
    
    def compute_log_likelihood(self, x, response):
        """
        Compute the value of log likelihood in each step 
        
        Argument:
        x -- training dataset, with the shape of [n_samples, n_features]
        
        Return:
        likelihood -- the current value of log likelihood
        
        """
        # Initialize the parameter
        likelihood= 0
        # Iterations: calculate step by step
        for k in range(self.k_components):
            n_k = np.sum(response[:,k])
            a_k = self.alpha[k]
            rv = stats.multivariate_normal(self.means[k], self.covs[k])
            f = np.dot(response[:,k], np.log(rv.pdf(x)))
            
            likelihood += n_k * np.log(a_k) + f
        
        return likelihood
